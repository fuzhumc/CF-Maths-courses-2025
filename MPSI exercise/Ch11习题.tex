为了解决习题,我们可以利用如下事实:若\(f\)在\([a,b]\)上连续,则\(\int_{[a,b]}f=F(b)-F(a)\),其中\(F\)为\(f\)的一个原函数.有时记\(\int_a^b f(x)\mathrm{d}x\)或\(\int_a^b f(t)\mathrm{d}t\)表示\(f\)在\([a,b]\)上的积分.

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  设\(f\in\mathcal{CM}([0,1],\mathbb{R})\)满足\(f([0,1])\subseteq[a,b]\)且\(\int_{[a,b]}f=0\).证明\(\int_{[0,1]}f^2\leqslant-ab\).\emph{考虑}\((f-a)(b-f)\).
\item
  \textbf{中值公式}
  设\(f\in\mathcal{C}([a,b],\mathbb{R}),g\in\mathcal{CM}([a,b],\mathbb{R}_+)\).

  \begin{enumerate}
  \def\labelenumii{\arabic{enumii}.}
  \tightlist
  \item
    证明:\[\exists c \in [a,b],\int_{[a,b]}fg=f(c)\int_{[a,b]}g\]
  \item
    设\(g \in \mathcal{CM}([a,b],\mathbb{R}_+^*)\).证明:\[\exists c \in (a,b),\int_{[a,b]}fg=f(c)\int_{[a,b]}g\]
  \end{enumerate}
\item
  \begin{enumerate}
  \def\labelenumii{\arabic{enumii}.}
  \tightlist
  \item
    设\(f\in\mathcal{C}([a,b],\mathbb{R})\).证明:\(\left\lvert \int_{[a,b]}f \right\rvert = \int_{[a,b]}\left\lvert f \right\rvert\)当且仅当\(f\)有一个固定的符号.
  \item
    设\(f\in\mathcal{C}([a,b],\mathbb{C})\).给出\(\left\lvert \int_{[a,b]}f \right\rvert = \int_{[a,b]}\left\lvert f \right\rvert\)的一个充要条件.\emph{我们回到情形}\(\int_{[a,b]} f=\left\lvert \int_{[a,b]} f \right\rvert\)\emph{开始}.
  \end{enumerate}
\item
  设\(f\in\mathcal{C}([a,b],\mathcal{R}_+)\).证明:\[\lim\limits_{n\to+\infty}\left(\int_{[a,b]}f^n\right)^\frac{1}{n}=\sup\limits_{x\in[a,b]}\left\lvert f(x) \right\rvert\]
\item
  设\(f\in\mathcal{CM}([0,1],\mathbb{R})\).对于自然数\(n\),记\(I_n=\int_{[a,b]}\left\lvert f \right\rvert^n\).证明\(I_n^2 \leqslant I_{n-1}I_{n+1}\).
\item
  设\(f\in\mathcal{C}([0,1],\mathbb{R})\),满足\(f(1)=0\).

  \begin{enumerate}
  \def\labelenumii{\arabic{enumii}.}
  \tightlist
  \item
    证明:对于任意\(\alpha\in[0,1)\),有\(\lim\limits_{n\to\infty}n\int_{0}^{\alpha}t^nf(t)\mathrm{d}t=0\).
  \item
    给出\(\lim\limits_{n\to\infty}n\int_0^1t^nf(t)\mathrm{d}t\).
  \end{enumerate}
\item
  设\(\mathcal{E}\)为\([a,b]\)上的严格正连续函数的全体.对于\(f\in\mathcal{E}\),记\[\varphi(f)=\left( \int_{[a,b]}f \right)\left( \int_{[a,b]}\frac{1}{f} \right)\]确定(若存在)\(\inf\limits_{f\in\mathcal{E}}\varphi(f)\).
\item
  设序列\[u_n=((n+1)(n+2)\cdots(n+n))^\frac{1}{n}\]找出\((u_n)\)的等价序列.
\item
  设\([a,b]\)为区间.证明:定义在\([a,b]\)上的函数\(f\)为阶梯函数当且仅当存在\(n\in\mathbb{N}^*,(\lambda_1,\lambda_2,\cdots,\lambda_n)\)以及\([a,b]\)的子区间\(I_1,I_2,\cdots,I_n\),使得\(f=\sum\limits_{k=0}^n\lambda_k\mathbf{1}_{I_k}\).
\item
  \begin{enumerate}
  \def\labelenumii{\arabic{enumii}.}
  \tightlist
  \item
    设\(f\in\mathcal{C}^1([a,b],\mathbb{C})\).证明:\[\lim\limits_{\lambda\to+\infty}\int_a^bf(t)e^{i\lambda t}\mathrm{d}t=0\]\emph{使用分部积分}.
  \item
    设\(f\in\mathcal{E}([a,b],\mathbb{C})\).证明:\[\lim\limits_{\lambda\to+\infty}\int_a^bf(t)e^{i\lambda t}\mathrm{d}t=0\]\emph{从}\(f\)\emph{为}\([a,b]\)\emph{的一个子区间的特征函数开始}.
  \item
    设\(f\in\mathcal{CM}([a,b],\mathbb{C})\).证明:\[\lim\limits_{\lambda\to+\infty}\int_a^bf(t)e^{i\lambda t}\mathrm{d}t=0\]
  \end{enumerate}
\item
  设\(f\in\mathcal{CM}([a,b],\mathbb{R})\).研究\[\lim\limits_{n\to\infty}\int_a^bf(t)\]
\end{enumerate}
