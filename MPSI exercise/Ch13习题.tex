  \begin{enumerate}
  \def\labelenumii{\arabic{enumii}.}
  \tightlist

  \item
    \(\lim\limits_{x\to0}\dfrac{(1-e^x)(1-\cos x)}{3x^3+2x^4}\).
  \item
    \(\lim\limits_{x\to0}(1+\sin x)^{\frac{1}{x}}\).
  \end{enumerate}

  \def\labelenumii{\arabic{enumii}.}
  \tightlist

  \end{enumerate}

  \def\labelenumii{\arabic{enumii}.}
  \tightlist
  \item

  \item
    求\(x_n,y_n\)的等价式.
  \end{enumerate}
